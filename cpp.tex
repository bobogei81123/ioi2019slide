\begin{frame}[fragile]{\btitle{Increase stack}}
  \begin{minted}{cpp}
#include <sys/resource.h>
void increase_stack_size() {
    const rlim_t ks = 64*1024*1024;
    struct rlimit rl;
    int res = getrlimit(RLIMIT_STACK, &rl);
    if (res == 0){
        if (rl.rlim_cur < ks){
            rl.rlim_cur = ks;
            res=setrlimit(RLIMIT_STACK, &rl);
        }
    }
}
  \end{minted}
